00135         // Remove an element from the List. O(n)
00136         //-------------------------------------------------------------------------
00137         template<class T>
00138         bool List<T>::remove(const T& data)
00139         {
00140             //If the root don't exist. Return false.
00141             if(!m_root)
00142                 return false;
00143 
00144             //If the root node is what we're looking for.
00145             if(m_root->data == data) {
00146                 Node<T> *tmp = m_root;
00147                 m_root = tmp->next;
00148                 free(tmp);
00149                 --m_size;
00150                 return true;
00151             }
00152 
00153             Node<T> *now = m_root->next;
00154             Node<T> *last = m_root;
00155 
00156             //While we have not found the data, and we're not at the end of the
00157             //list, go to next element.
00158             while(now && now->data != data) {
00159                 last = now;
00160                 now = now->next;
00161             }
00162 
00163             //We're at the end of the list, so no such element in the list.
00164             if(!now)
00165                 return false;
00166 
00167             last->next = now->next;
00168             free(now);
