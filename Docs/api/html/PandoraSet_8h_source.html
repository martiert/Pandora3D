00106         //-------------------------------------------------------------------------
00107         // Appends an element to the first free place in the set. If no free spots,
00108         // we have to resize the array.
00109         //-------------------------------------------------------------------------
00110         template<class T>
00111         void Set<T>::insert(const T& in)
00112         {
00113             //If we have room for the element, no rearranging have to be done.
00114             if(m_elements < m_size) {
00115                 m_data[m_elements++] = in;
00116             } else {
00117                 //Checks if we are allowed to resize the set.
00118                 assert(m_increment != 0 && 
00119                         "Not allowed to resize this set, but resize needed");
00120 
00121                 //Resize the array
00122                 T *tmp = new T[m_size + m_increment];
00123 
00124                 //Copy the data to the new array
00125                 for(int i = 0; i < m_size; ++i)
00126                     tmp[i] = m_data[i];
00127 
00128                 //Delete the old array, copy the new large array to the data 
00129                 //pointer and set the new size to the old size + the 
00130                 //incrementation.
00131                 delete[] m_data;
00132                 m_data = tmp;
00133                 m_size += m_increment;
00134 
00135                 //Set the data in the set
00136                 m_data[m_elements++] = in;
00137             }
00138         }

00157         //-------------------------------------------------------------------------
00158         // Remove an element from the set. Returns true if the element is removed,
00159         // returns false otherwise
00160         //-------------------------------------------------------------------------
00161         template<class T>
00162         bool Set<T>::remove(const T& elem)
00163         {
00164             size_t i = 0;
00165 
00166             //Search for the element.
00167             while(i < m_elements && m_data[i] != elem) {
00168                 ++i;
00169             }
00170 
00171             //The element doesn't exist.
00172             if(i == m_elements)
00173                 return false;
00174 
00175             //Shuffle all the remaining data
00176             for(;i < m_size - 1; ++i)
00177                 m_data[i] = m_data[i + 1];
00178             m_data[i] = NULL;
00179 
00180             return true;
00181         }
