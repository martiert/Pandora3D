00127         // Insert a value into the HashTable. Returns false if the table is full or
00128         //          if it hashes to the same value as another element.
00129         //-------------------------------------------------------------------------
00130         template<class Key, class Value>
00131         bool HashTable<Key,Value>::insert(const Key& key, const Value& value)
00132         {
00133             //No spaces free.
00134             if(m_elements == m_size) {
00135                 return false;
00136             }
00137 
00138             //Get the index for this element.
00139             int index = hashFunction(key);
00140 
00141             //Crashes with other element
00142             if(m_values[index]) {
00143                 return false;
00144             }
00145 
00146             //Successful insertion.
00147             m_values[index] = malloc(sizeof(Node<Key,Value>));
00148             m_values[index]->key = key;
00149             m_values[index]->value = value;
00150             m_elements++;
00151             return true;

00226         //---------------------------------------------------------------------
00227         //                      The Hash function.
00228         //---------------------------------------------------------------------
00229         template<class Key, class Value>
00230         int HashTable<Key,Value>::hashFunction(const Key& key) const
00231         {
00232             if(userHashFunction)
00233                 return (*userHashFunction)(key);
00234 
00235             static double multiply = 0.5*(sqrt(5.0) - 1.0);
00236             size_t hashKey;
00237             memcpy(&hashKey, &key, sizeof(size_t));
00238             hashKey %= m_size;
00239             double fraction = fmod(multiply * hashKey, 1.0);
